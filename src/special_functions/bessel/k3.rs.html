use crate::approximations::polynomial;

pub fn lower(x: f64) -> f64 {
    let x2 = x.powi(2);

    x.powi(-3)
        * polynomial(
            x2,
            &[
                8.,
                -1.,
                0.125,
                -0.02151246213177248,
                -0.002158330966569113,
                -0.0000653515033308945,
                -1.020687835614275e-6,
            ],
        )
        + x.powi(3)
            * x.ln()
            * polynomial(
                x2,
                &[
                    0.02083333333333333,
                    0.001302083333333333,
                    0.00003255208333333333,
                    4.521122685185185e-7,
                ],
            )
}

                8.151741880267863,

        )
}

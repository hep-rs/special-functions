    x.powi(4)

pub fn upper(x: f64) -> f64 {
    let x1 = x.recip();
    let amp = x1
        * polynomial(
            x.powi(-2),
            &[
                0.6366197723675813,
                5.013380707394703,
                51.70048854500788,
                420.066469428189,
                1378.343102811245,
                -5272.162368253013,
                68867.62093530498,
                -1.678648260298059e6,
                6.334274294718456e7,
                -3.36508321906918e9,
            ],
        );
    let amp = amp.sqrt();
    let phase = x - 7.068583470577035
        + x1 * polynomial(
            x1,
            &[
                7.875,
                0.0,
                6.3984375,
                0.0,
                -26.1720703125,
                0.0,
                -508.4969787597656,
            ],
        );

    amp * phase.cos()
}

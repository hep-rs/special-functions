use crate::approximations::polynomial;

pub fn lower(x: f64) -> f64 {
    let x2 = x.powi(2);

    x.powi(-2)
        * polynomial(
            x2,
            &[
                2.,
                -0.5,
                0.1082414394573016,
                0.01596456439921957,
                0.0006209629499756117,
                0.00001179614175885279,
                1.346502336473863e-7,
            ],
        )
        + x.powi(2)
            * x.ln()
            * polynomial(
                x2,
                &[
                    -0.125,
                    -0.01041666666666667,
                    -0.0003255208333333333,
                    -5.425347222222222e-6,
                    -5.651403356481481e-8,
                ],
            )
}

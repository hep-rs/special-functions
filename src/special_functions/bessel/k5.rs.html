    x.powi(-5)
        * polynomial(
            x2,
            &[
                384.,
                -24.,
                1.,
                -0.04166666666666667,
                0.002604166666666667,
                -0.000327499526647156,
                -0.00001997538536955743,
                -4.189840818543418e-7,

            * x.ln()

                x2,

                    0.0002604166666666667,

pub fn lower(x: f64) -> f64 {
    let x2 = x.powi(2);

    x.powi(2)
        * polynomial(
            x2,
            &[
                0.125,
                -0.01041666666666667,
                0.0003255208333333333,
                -5.425347222222222e-6,
                5.651403356481481e-8,
                -4.036716683201058e-10,
            ],
        )
}

pub fn upper(x: f64) -> f64 {
    let x1 = x.recip();
    let amp = x1
        * polynomial(
            x.powi(-2),
            &[
                0.6366197723675813,
                1.193662073189215,
                1.566681471060845,
                -2.937527758239084,
                21.20527850478839,
                -310.1271981325302,
                7462.435705064007,
                -265049.7253102198,
                1.29832951382428e7,
                -8.368815657859004e8,
            ],
        );
    let amp = amp.sqrt();
    let phase = x - 3.926990816987242
        + x1 * polynomial(
            x1,
            &[
                1.875,
                0.0,
                -0.3515625,
                0.0,
                -1.4501953125,
                0.0,
                8.307429722377232,
            ],
        );

    amp * phase.cos()
}

pub fn lower(x: f64) -> f64 {
    let x2 = x.powi(2);

    x.powi(3)
        * polynomial(
            x2,
            &[
                0.02083333333333333,
                -0.001302083333333333,
                0.00003255208333333333,
                -4.521122685185185e-7,
                4.036716683201058e-9,
                -2.522947927000661e-11,
            ],
        )
}

pub fn upper(x: f64) -> f64 {
    let x1 = x.recip();
    let amp = x1
        * polynomial(
            x.powi(-2),
            &[
                0.6366197723675813,
                2.785211504108168,
                14.1001332395476,
                32.31280534062992,
                -91.8895401874163,
                930.38159439759,
                -18123.05814086973,
                559549.4200993529,
                -2.478629071846352e7,
                1.480636616390439e9,
            ],
        );
    let amp = amp.sqrt();
    let phase = x - 5.497787143782138
        + x1 * polynomial(
            x1,
            &[
                4.375,
                0.0,
                1.002604166666667,
                0.0,
                -11.8603515625,
                0.0,
                -24.79629516601562,
            ],
        );

    amp * phase.cos()
}

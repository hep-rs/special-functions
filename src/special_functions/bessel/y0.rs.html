pub fn lower(x: f64) -> f64 {
    let x2 = x.powi(2);

    x.powi(-0)
        * polynomial(
            x2,
            &[
                -0.07380429510868723,
                0.1776060168690671,
                -0.01607396802593843,
                0.000538602666861655,
                -9.49500520522155e-6,
                1.03584760336281e-7,
            ],
        )
        + x.powi(0)
            * x.ln()
            * polynomial(
                x2,
                &[
                    0.6366197723675813,
                    -0.1591549430918953,
                    0.00994718394324346,
                    -0.0002763106650900961,
                    4.317354142032751e-6,
                    -4.317354142032751e-8,
                ],
            )
}

pub fn upper(x: f64) -> f64 {
    let x1 = x.recip();
    let amp = x1
        * polynomial(
            x.powi(-2),
            &[
                0.6366197723675813,
                -0.07957747154594767,
                0.1342869832337867,
                -0.6994113710093057,
                7.496815633005995,
                -136.6294649115343,
                3788.621204109419,
                -148635.7283112213,
                7.838212235162061e6,
                -5.34849120990989e8,
            ],
        );
    let amp = amp.sqrt();
    let phase = x - 0.7853981633974483
        + x1 * polynomial(
            x1,
            &[
                -0.125,
                0.0,
                0.06510416666666667,
                0.0,
                -0.2095703125,
                0.0,
                1.638065883091518,
            ],
        );

    amp * phase.sin()
}

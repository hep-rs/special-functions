use crate::approximations::polynomial;

pub fn lower(x: f64) -> f64 {
    let x2 = x.powi(2);

    x.powi(0)
        * polynomial(
            x2,
            &[
                1.,
                -0.25,
                0.015625,
                -0.0004340277777777778,
                6.781684027777778e-6,
                -6.781684027777778e-8,
            ],
        )
}

pub fn upper(x: f64) -> f64 {
    let x1 = x.recip();
    let amp = x1
        * polynomial(
            x.powi(-2),
            &[
                0.6366197723675813,
                -0.07957747154594767,
                0.1342869832337867,
                -0.6994113710093057,
                7.496815633005995,
                -136.6294649115343,
                3788.621204109419,
                -148635.7283112213,
                7.838212235162061e6,
                -5.34849120990989e8,
            ],
        );
    let amp = amp.sqrt();
    let phase = x - 0.7853981633974483
        + x1 * polynomial(
            x1,
            &[
                -0.125,
                0.0,
                0.06510416666666667,
                0.0,
                -0.2095703125,
                0.0,
                1.638065883091518,
            ],
        );

    amp * phase.cos()
}

use crate::approximations::polynomial;

pub fn lower(x: f64) -> f64 {
    let x2 = x.powi(2);

    x.powi(-0)
        * polynomial(
            x2,
            &[
                0.1159315156584124,
                0.2789828789146031,
                0.02524892993216269,
                0.000846035090708223,
                0.00001491471929926043,
                1.627105610481598e-7,
            ],
        )
        + x.powi(0)
            * x.ln()
            * polynomial(
                x2,
                &[
                    -1.,
                    -0.25,
                    -0.015625,
                    -0.0004340277777777778,
                    -6.781684027777778e-6,
                    -6.781684027777778e-8,
                ],
            )
}

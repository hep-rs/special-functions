#![allow(clippy::all)]

use crate::approximations::polynomial;

pub fn lower(x: f64) -> f64 {
    let x2 = x.powi(2);
    let x6 = x2.powi(3);
    let ln = x.ln();
    let ln2 = ln.powi(2);

    x * (polynomial(
        x2,
        &[
            0.5,
            -0.02898287891460311,
            -0.07699143945730156,
            -0.02399151808069941,
            -0.001811206658024423,
        ],
    ) + x2
        * ln
        * polynomial(
            x2,
            &[0.25, 0.125, 0.01981464013567461, -0.008661410998048935],
        )
        + x6 * ln2 * polynomial(x2, &[0.015625, 0.01302083333333333]))
}

pub fn upper(x: f64) -> f64 {
    polynomial(
        x.recip(),
        &[
            1.,
            -1.5,
            1.875,
            -1.875,
            1.0546875,
            1.40625,
            -7.2509765625,
            21.09375,
            -58.15200805664062,
            177.978515625,
            -645.6587791442871,
        ],
    )
}

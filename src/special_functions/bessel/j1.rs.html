use crate::approximations::polynomial;

pub fn lower(x: f64) -> f64 {
    let x2 = x.powi(2);

    x.powi(1)
        * polynomial(
            x2,
            &[
                0.5,
                -0.0625,
                0.002604166666666667,
                -0.00005425347222222222,
                6.781684027777778e-7,
                -5.651403356481481e-9,
            ],
        )
}

pub fn upper(x: f64) -> f64 {
    let x1 = x.recip();
    let amp = x1
        * polynomial(
            x.powi(-2),
            &[
                0.6366197723675813,
                0.238732414637843,
                -0.2238116387229778,
                0.979175919413028,
                -9.63876295672199,
                166.9915682252085,
                -4477.461423038404,
                171502.7634360246,
                -8.883307199850336e6,
                5.977725469899289e8,
            ],
        );
    let amp = amp.sqrt();
    let phase = x - 2.356194490192345
        + x1 * polynomial(
            x1,
            &[
                0.375,
                0.0,
                -0.1640625,
                0.0,
                0.3708984375,
                0.0,
                -2.369397844587054,
            ],
        );

    amp * phase.cos()
}
